CREATE TABLE tblTasks (
    TaskID INT IDENTITY(1,1) PRIMARY KEY,
    TaskUUID NVARCHAR(200) NOT NULL UNIQUE DEFAULT NEWID(),
    TaskTitle NVARCHAR(100) NOT NULL,
    TaskDescription NVARCHAR(MAX) NOT NULL,
    TaskType NVARCHAR(50) NOT NULL,
    Status INT NOT NULL,  -- 0=Received, 1=Completed, 2=Failed
    DateCreatedFrom DATE NOT NULL,
    DateCreatedTo DATE NULL,
    UserAssigned INT NOT NULL,  -- Adjust table/column name as needed
    ValidityFrom DATE NOT NULL DEFAULT GETDATE(),
    ValidityTo DATE NULL
);
